git init 			(criar repositorio)
git add .			(adicionar arquivos no repositorio)
git commit -m "mensagem"	(gravar informação no repositorio)
git remote "link"		( )
git push -u origin		( )
git status			(as informações adicionadas no repositorio)
git revert 			(reverter o ultimo commit)
git log				(visualizar histórico dos commit)
git branch			(listar as branchs que exitem)
git checkout -b nome_branch	(criar nova branch)
git checkout master		(usar branch)
git branch nome_branch -d	(deleta a branch)
git pull // git fetch		(atualizar o trabalho loca com a versão recente do repositório remoto)
git merge nome_outra_branch	(juntar duas branches, com diferentes commits)
git revert <commit>		(desfaz o commit do repositório remoto)
git reset			(reseta)
